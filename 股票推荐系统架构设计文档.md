## 1. 项目概述

### 1.1 项目背景

本项目旨在构建一个基于量化策略的股票推荐系统，通过收集和分析市场涨停数据，识别具有潜在投资价值的股票。系统聚焦于市场热点捕捉和强者恒强策略，为投资者提供数据驱动的决策支持。

### 1.2 核心功能

| 功能模块 | 描述 |
|---------|------|
| **涨停数据采集** | 每日自动收集涨停股票信息，包括涨停价格、时间、原因、题材等 |
| **连板追踪** | 识别过去10个交易日内连续涨停的股票 |
| **量能筛选** | 筛选前一日涨停且次日成交量放大超过2倍的股票 |
| **价格突破回测** | 回测当前价格突破前15个交易日涨停价的股票表现 |

### 1.3 系统目标

- **数据准确性**：确保涨停数据的准确性和时效性
- **策略有效性**：通过历史回测验证策略有效性
- **系统稳定性**：保障系统7x24小时稳定运行
- **扩展性**：支持新策略的快速接入和配置

##### 策略2：连板追踪策略

```python
class ConsecutiveLimitUpStrategy:
    """
    功能：识别过去10个交易日连续涨停的股票
    输入：当前日期
    输出：连板股票列表
    """
    def execute(self, current_date: date) -> List[ConsecutiveStock]:
        # 1. 获取过去10个交易日
        trading_days = self.get_trading_days(
            current_date,
            lookback_days=10
        )

        # 2. 计算每个股票的连板情况
        stocks = self.db.query_stocks()
        results = []

        for stock in stocks:
            consecutive_count = 0
            limit_up_records = []

            for day in trading_days:
                if self.is_limit_up(stock.code, day):
                    consecutive_count += 1
                    limit_up_records.append(day)
                else:
                    break  # 连续中断

            if consecutive_count >= 2:
                results.append(ConsecutiveStock(
                    code=stock.code,
                    name=stock.name,
                    consecutive_days=consecutive_count,
                    start_date=limit_up_records[-1],
                    latest_date=limit_up_records[0]
                ))

        # 3. 按连板天数排序
        results.sort(key=lambda x: x.consecutive_days, reverse=True)
        return results
```

##### 策略3：量能筛选策略

```python
class VolumeSurgeStrategy:
    """
    功能：筛选前一日涨停且次日成交量/前日成交量 > 2 的股票
    输入：当前日期
    输出：量能异动股票列表
    """
    def execute(self, current_date: date) -> List[VolumeSurgeStock]:
        results = []

        # 1. 获取前一日的涨停股票
        prev_date = self.get_prev_trading_day(current_date)
        limit_up_stocks = self.db.query_limit_up(prev_date)

        # 2. 计算量比
        for stock in limit_up_stocks:
            # 前一日涨停成交量
            volume_limit_up = stock.volume

            # 当日成交量
            volume_today = self.get_volume(stock.code, current_date)

            # 计算量比
            volume_ratio = volume_today / volume_limit_up

            if volume_ratio > 2.0:
                results.append(VolumeSurgeStock(
                    code=stock.code,
                    name=stock.name,
                    limit_up_date=prev_date,
                    current_date=current_date,
                    volume_ratio=volume_ratio,
                    volume_today=volume_today,
                    volume_limit_up=volume_limit_up
                ))

        # 3. 按量比排序
        results.sort(key=lambda x: x.volume_ratio, reverse=True)
        return results
```

##### 策略4：价格突破回测

```python
class PriceBreakoutBacktest:
    """
    功能：回测当前股价 > 前15个交易日涨停价的股票
    输入：当前日期
    输出：突破股票列表
    """
    def execute(self, current_date: date) -> List[BreakoutStock]:
        results = []

        # 1. 获取前15个交易日
        trading_days = self.get_trading_days(
            current_date,
            lookback_days=15
        )

        # 2. 获取期间涨停股票
        limit_up_records = self.db.query_limit_up_range(
            start_date=trading_days[-1],
            end_date=trading_days[0]
        )

        # 3. 检查价格突破
        for record in limit_up_records:
            # 获取当前价格
            current_price = self.get_current_price(
                record.stock_code,
                current_date
            )

            # 判断是否突破
            if current_price > record.limit_up_price:
                results.append(BreakoutStock(
                    code=record.stock_code,
                    name=record.stock_name,
                    limit_up_date=record.trade_date,
                    limit_up_price=record.limit_up_price,
                    current_price=current_price,
                    breakout_ratio=current_price / record.limit_up_price - 1
                ))

        # 4. 按突破幅度排序
        results.sort(key=lambda x: x.breakout_ratio, reverse=True)
        return results
```

#### 4.3.3 回测流程

```python
class BacktestEngine:
    """回测引擎"""

    def run(self, config: BacktestConfig) -> BacktestResult:
        """
        执行回测
        """
        # 1. 加载历史数据
        self.load_historical_data(config.start_date, config.end_date)

        # 2. 初始化账户
        account = Account(config.initial_capital)

        # 3. 遍历交易日
        for trade_date in self.trading_days:
            # 3.1 获取策略信号
            signals = self.generate_signals(trade_date)

            # 3.2 执行交易
            self.execute_trades(account, signals, trade_date)

            # 3.3 更新持仓价值
            self.update_portfolio_value(account, trade_date)

            # 3.4 风险控制
            self.apply_risk_control(account, trade_date)

        # 4. 计算绩效指标
        metrics = self.calculate_metrics(account)

        # 5. 生成报告
        report = self.generate_report(account, metrics)

        return BacktestResult(
            config=config,
            account=account,
            metrics=metrics,
            report=report
        )
```

```python
class NotificationService:
    """通知服务"""

    def __init__(self):
        self.channels = self.init_channels()

    def send(self, level: NotificationLevel, message: str,
             targets: List[str] = None):
        """
        发送通知

        Args:
            level: 通知级别
            message: 通知内容
            targets: 接收者列表
        """
        # 1. 根据级别选择渠道
        channels = self.get_channels_by_level(level)

        # 2. 智能聚合
        message = self.aggregate_message(message)

        # 3. 发送通知
        for channel in channels:
            try:
                channel.send(message, targets)
            except Exception as e:
                logger.error(f"发送失败: {channel.name}, {e}")

        # 4. 记录发送历史
        self.log_notification(level, message, channels)
```
